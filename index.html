var t = 0;
var cellSize = 50;
var hCellSize = cellSize / 2;
function setup() {
  createCanvas(size, size);
  colorMode(HSB, 100);
  angleMode(DEGREES);
  rectMode(CENTER);
  noStroke();
  for(var x = hCellSize; x < size; x+=cellSize) {
    for(var y = hCellSize; y < size; y+=cellSize) {
      push();
      translate(x, y);
      rotate(map(x + y, 0, size * 2, 0, 360));
      fill(map(x + y, 0, size * 2, 0, 10), 50, 100);
      rect(0, 0, hCellSize, hCellSize, 2);
      pop()
    }
  }
}

    pop();
  }
